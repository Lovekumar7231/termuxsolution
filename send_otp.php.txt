<?php
session_start();
$data = json_decode(file_get_contents("php://input"), true);

$email = $data['email'];
$otp = rand(100000, 999999);

$_SESSION['otp'] = $otp;
$_SESSION['otp_time'] = time();

mail(
  $email,
  "OTP Verification",
  "Your OTP is: $otp (valid 5 min)",
  "From: noreply@yourdomain.com"
);

echo "OTP Sent";
