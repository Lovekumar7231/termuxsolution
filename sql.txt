-- 1. Users Table
CREATE TABLE IF NOT EXISTS users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    role VARCHAR(20) DEFAULT 'user', -- 'admin', 'prime', 'user'
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 2. Tools Table (Updated with Image & Video)
CREATE TABLE IF NOT EXISTS termux_tools (
    id INT AUTO_INCREMENT PRIMARY KEY,
    tool_name VARCHAR(255) NOT NULL,
    tool_commands TEXT NOT NULL,
    tool_img VARCHAR(255) DEFAULT 'https://via.placeholder.com/300',
    tool_video VARCHAR(255) DEFAULT ''
);
